    <script>

        const words = document.querySelectorAll('.word');

        let currentIndex = 0;

        let utterance;


        function startReading() {

            const text = Array.from(words).map(word => word.textContent).join(' ');

            currentIndex = 0;

            resetHighlighting();


            utterance = new SpeechSynthesisUtterance(text);

            utterance.rate = 0.8;  // Slower rate for beginners


            utterance.onboundary = (event) => {

                if (event.name === 'word') {

                    highlightWord(currentIndex);

                    currentIndex++;

                }

            };


            utterance.onend = () => resetHighlighting();


            speechSynthesis.speak(utterance);

        }


        function highlightWord(index) {

            resetHighlighting();

            if (index < words.length) {

                words[index].classList.add('highlight');

            }

        }


        function resetHighlighting() {

            words.forEach(word => word.classList.remove('highlight'));

        }

    </script>
